int vis[21];
vector<int> v;
bool check(int n, int a, int b, int c, int d) {

	if (a == b && b == c && c == d && a != 0) {
		return true;
	}

	bool f = false;
	for (int i = 0; i < n; i++) {
		if (vis[i] != 1) {
			vis[i] = 1;
			a += v[i];
			f = check(n, a, b, c, d);
			vis[i] = 0;
			a -= v[i];
		}
		if (f == true)
					return true;
		if (vis[i] != 1) {
			vis[i] = 1;
			b += v[i];
			f = check(n, a, b, c, d);
			vis[i] = 0;
			b -= v[i];
		}if (f == true)
			return true;
		if (vis[i] != 1) {
			vis[i] = 1;
			c += v[i];
			f = check(n, a, b, c, d);
			vis[i] = 0;
			c -= v[i];
		}if (f == true)
			return true;
		if (vis[i] != 1) {
			vis[i] = 1;
			d += v[i];
			f = check(n, a, b, c, d);
			vis[i] = 0;
			d -= v[i];
		}

		if (f == true)
			return true;
	}

		return false;


}

int main() {

	//freopen("in.txt", "r", stdin);

	int t;
	scanf("%d", &t);
	while (t--) {
		int n;
		scanf("%d", &n);
		v.resize(n);
		memset(vis, 0, sizeof(vis));
		for (int i = 0; i < n; i++) {
			scanf("%d", &v[i]);
		}

		bool ans = check(n, 0, 0, 0, 0);

		if (ans == true) {
			printf("yes\n");
		} else
			printf("no\n");
	}
	return 0;
}
